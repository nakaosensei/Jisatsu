function enviarTrace()
{
	var limit = 4500;
	var postStack = '';
	var stacktrace = '';
	
	if (stacktrace.length<=limit) 
	{
		postStack = stacktrace;
	}
	else
	{
		postStack = stacktrace.substring(0,limit-1);
	}
	
	var urlSalvaSessao = "/ForwardAction.do?applicationName=ControleAcesso&modulo=&path=/PrincipalUC/ControleAcessoInserirTrace.do";
	var info = '?descricao=' + $('#idDescricao').val() + '@AND@stackTrace=' + postStack + '@AND@contexto=reflora';

	
	
	$('#idDivStackTrace').hide();
	$('#reportTrace').hide();

	$.ajax({
		type:"post",
		url: getPath() + urlSalvaSessao + info ,
		success: function(data) {
				var id = $(data).find('id').text();
				if(id != 'null') {
					$('#messages').html('Reportado com sucesso. Id: ' + id);
				}
				else {
					$('#messages').html('Falha na hora de reportar. Entre em contato com o administrador do sistema.')
				}
			}
		});
}
